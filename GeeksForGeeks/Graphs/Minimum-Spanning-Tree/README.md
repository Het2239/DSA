class DSU{
    vector<int> rank;
    vector<int> parent;
    public:
    DSU(int n){
        rank.resize(n+1,0);
        parent.resize(n+1);
        for(int i=0;i<=n;i++){
            parent[i]=i;
        }
    }
    
    int findUPar(int node){
        if(parent[node]==node){
            return node;
            
        }
        return parent[node]= findUPar(parent[node]);
    }
    
    void unionByRank(int u,int v){
        int paru = findUPar(u);
        int parv = findUPar(v);
        if(paru==parv) return;
        if(rank[paru]<rank[parv]) parent[paru] = parv;
        else if(rank[parv]<rank[paru]) parent[parv] = paru;
        else{
            parent[parv]=paru;
            rank[paru]++;
        }
    }
};
